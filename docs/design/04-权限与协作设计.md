## 设计文档：权限与协作

版本：v1.0 | 日期：2026-06-11

---

### 一、现状分析

当前权限系统非常简单：

1. **RBAC 只有两个角色**：admin 和 user，通过 `Role` + `UserRole` 表关联
2. **知识库访问控制粗粒度**：`KnowledgeBase.createdBy` + `isPublic` 两个字段控制，要么是创建者要么公开
3. **无协作概念**：没有知识库成员、团队、邀请等机制
4. **无审批流程**：文档从创建到发布没有审核环节
5. **无评论系统**：不能对文档添加评论或批注
6. **访问控制分散**：各 Service 各自实现 `OR: [{ createdBy: userId }, { isPublic: true }]` 的过滤逻辑，不统一

---

### 二、目标

1. 知识库成员管理：邀请协作者，设置知识库级读/写/管理权限
2. 细粒度权限：从 admin/user 两级扩展为资源级权限控制
3. 文档审批工作流：支持创建→审核→发布的流程
4. 评论与批注：支持文档级和段落级评论
5. 统一访问控制层：抽出通用的权限判断逻辑

#### 4.1 统一访问控制层（AccessControlService）

抽取分散在各 Service 中的权限逻辑为统一服务：

```
src/common/services/
├── access-control.service.ts
```

```typescript
// access-control.service.ts
@Injectable()
export class AccessControlService {
  constructor(private prisma: PrismaService) {}

  /** 检查用户是否可访问指定知识库 */
  async canAccessKb(userId: string, kbId: string): Promise<boolean> {
    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: kbId } });
    if (!kb) return false;
    if (kb.isPublic) return true;
    if (kb.createdBy === userId) return true;
    // 检查成员关系
    const member = await this.prisma.kbMember.findUnique({
      where: { kbId_userId: { kbId, userId } },
    });
    return !!member;
  }

  /** 获取用户在知识库中的角色 */
  async getKbRole(userId: string, kbId: string): Promise<KbRole | null> {
    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: kbId } });
    if (kb?.createdBy === userId) return 'ADMIN'; // 创建者等同 ADMIN

    const member = await this.prisma.kbMember.findUnique({
      where: { kbId_userId: { kbId, userId } },
    });
    return member?.role || null;
  }

  /** 检查用户是否可编辑知识库中的文档 */
  async canEditDocument(userId: string, kbId: string): Promise<boolean> {
    const role = await this.getKbRole(userId, kbId);
    return role === 'EDITOR' || role === 'ADMIN';
  }

  /** 检查用户是否可管理知识库（成员、设置） */
  async canManageKb(userId: string, kbId: string): Promise<boolean> {
    const role = await this.getKbRole(userId, kbId);
    return role === 'ADMIN';
  }

  /** 构建知识库访问过滤条件（替代各 Service 中的手写 OR 逻辑） */
  buildKbAccessFilter(userId: string): any {
    return {
      OR: [
        { isPublic: true },
        { createdBy: userId },
        { members: { some: { userId } } },
      ],
    };
  }

  /** 构建文档访问过滤条件 */
  buildDocAccessFilter(userId: string): any {
    return {
      OR: [
        { isPublic: true },
        { createdBy: userId },
        { kb: { isPublic: true } },
        { kb: { createdBy: userId } },
        { kb: { members: { some: { userId } } } },
      ],
    };
  }

  /** 检查是否是系统管理员 */
  async isSystemAdmin(userId: string): Promise<boolean> {
    const userRoles = await this.prisma.userRole.findMany({
      where: { userId },
      include: { role: true },
    });
    return userRoles.some(ur => ur.role.name === 'admin');
  }
}
```

// kb-members.service.ts
@Injectable()
export class KbMembersService {
  constructor(
    private prisma: PrismaService,
    private accessControl: AccessControlService,
  ) {}

  /** 获取知识库成员列表 */
  async getMembers(kbId: string) {
    return this.prisma.kbMember.findMany({
      where: { kbId },
      include: {
        user: { select: { id: true, nickname: true, phone: true, avatar: true, email: true } },
        inviter: { select: { id: true, nickname: true } },
      },
      orderBy: { joinedAt: 'asc' },
    });
  }

  /** 邀请成员（需要 ADMIN 权限） */
  async invite(kbId: string, inviterId: string, dto: {
    userId: string;       // 或 phone/email 查找
    role: KbRole;
  }) {
    // 权限检查
    const canManage = await this.accessControl.canManageKb(inviterId, kbId);
    if (!canManage) throw new ForbiddenException('无权限管理成员');

    // 不能邀请自己
    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: kbId } });
    if (kb?.createdBy === dto.userId) throw new BadRequestException('不能邀请知识库创建者');

    return this.prisma.kbMember.upsert({
      where: { kbId_userId: { kbId, userId: dto.userId } },
      create: { kbId, userId: dto.userId, role: dto.role, invitedBy: inviterId },
      update: { role: dto.role },
    });
  }

  /** 更新成员角色 */
  async updateRole(kbId: string, userId: string, role: KbRole, operatorId: string) {
    const canManage = await this.accessControl.canManageKb(operatorId, kbId);
    if (!canManage) throw new ForbiddenException('无权限管理成员');

    return this.prisma.kbMember.update({
      where: { kbId_userId: { kbId, userId } },
      data: { role },
    });
  }

  /** 移除成员 */
  async removeMember(kbId: string, userId: string, operatorId: string) {
    const canManage = await this.accessControl.canManageKb(operatorId, kbId);
    if (!canManage) throw new ForbiddenException('无权限管理成员');

    return this.prisma.kbMember.delete({
      where: { kbId_userId: { kbId, userId } },
    });
  }

  /** 主动离开知识库 */
  async leaveKb(kbId: string, userId: string) {
    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: kbId } });
    if (kb?.createdBy === userId) {
      throw new BadRequestException('创建者不能离开知识库，请先转让');
    }
    return this.prisma.kbMember.delete({
      where: { kbId_userId: { kbId, userId } },
    });
  }
}

**审批流程**：

```
文档编辑完成
  ↓
用户点击「提交审核」
  ↓
创建 DocumentReview 记录（status=PENDING）
文档状态 → PENDING_REVIEW（新增枚举值）
  ↓
知识库 ADMIN/EDITOR 在审核列表中看到待审文档
  ↓
审核人操作：
  ├─ 通过 → DocumentReview.status=APPROVED
  │         Document.reviewStatus=APPROVED
  │         自动触发发布（进入 ingestion 流水线）
  └─ 拒绝 → DocumentReview.status=REJECTED
            Document.reviewStatus=REJECTED
            附带拒绝原因（comment）
            通知作者修改
```

```typescript
// document-review.service.ts
@Injectable()
export class DocumentReviewService {
  constructor(
    private prisma: PrismaService,
    private accessControl: AccessControlService,
    private ingestionService: IngestionService,
  ) {}

  /** 提交文档审核 */
  async submitForReview(documentId: string, userId: string) {
    const doc = await this.prisma.document.findUnique({ where: { id: documentId } });
    if (!doc) throw new BadRequestException('文档不存在');

    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: doc.kbId } });
    if (!kb?.requireReview) {
      // 不需要审核，直接发布
      return this.directPublish(documentId);
    }

    // 创建审核记录
    const review = await this.prisma.documentReview.create({
      data: { documentId, reviewerId: userId, status: 'PENDING' },
    });

    await this.prisma.document.update({
      where: { id: documentId },
      data: { reviewStatus: 'PENDING' },
    });

    return review;
  }

  /** 审核通过 */
  async approve(reviewId: string, reviewerId: string, comment?: string) {
    const review = await this.prisma.documentReview.findUnique({
      where: { id: reviewId },
      include: { document: true },
    });
    if (!review) throw new BadRequestException('审核记录不存在');

    // 验证审核人权限
    const canManage = await this.accessControl.canManageKb(reviewerId, review.document.kbId);
    if (!canManage) throw new ForbiddenException('无审核权限');

    await this.prisma.documentReview.update({
      where: { id: reviewId },
      data: { status: 'APPROVED', comment, reviewerId, reviewedAt: new Date() },
    });

    await this.prisma.document.update({
      where: { id: review.documentId },
      data: { reviewStatus: 'APPROVED', publishedAt: new Date() },
    });

    // 自动触发索引
    await this.ingestionService.enqueueDocument(review.documentId);
  }

  /** 审核拒绝 */
  async reject(reviewId: string, reviewerId: string, comment: string) {
    // 类似逻辑，设置 REJECTED 状态
  }

  /** 获取待审核列表 */
  async getPendingReviews(kbId: string, reviewerId: string) {
    return this.prisma.documentReview.findMany({
      where: {
        document: { kbId },
        status: 'PENDING',
      },
      include: {
        document: { select: { id: true, title: true, type: true, updatedAt: true, createdBy: true } },
      },
      orderBy: { createdAt: 'asc' },
    });
  }

  /** 直接发布（无需审核） */
  private async directPublish(documentId: string) {
    await this.prisma.document.update({
      where: { id: documentId },
      data: { reviewStatus: 'APPROVED', publishedAt: new Date() },
    });
    await this.ingestionService.enqueueDocument(documentId);
  }
}
```

```typescript
// comments.service.ts
@Injectable()
export class CommentsService {
  constructor(
    private prisma: PrismaService,
    private accessControl: AccessControlService,
  ) {}

  /** 创建评论 */
  async create(userId: string, dto: {
    documentId: string;
    chunkId?: string;    // 段落批注
    parentId?: string;   // 回复
    content: string;
  }) {
    // 验证文档访问权限
    const doc = await this.prisma.document.findUnique({ where: { id: dto.documentId } });
    if (!doc) throw new BadRequestException('文档不存在');

    const kb = await this.prisma.knowledgeBase.findUnique({ where: { id: doc.kbId } });
    if (!kb?.allowComments) throw new BadRequestException('该知识库不允许评论');

    const canAccess = await this.accessControl.canAccessKb(userId, doc.kbId);
    if (!canAccess) throw new ForbiddenException('无访问权限');

    return this.prisma.comment.create({
      data: { ...dto, userId },
      include: { user: { select: { id: true, nickname: true, avatar: true } } },
    });
  }

  /** 获取文档评论（含嵌套回复） */
  async getComments(documentId: string, chunkId?: string) {
    const comments = await this.prisma.comment.findMany({
      where: {
        documentId,
        parentId: null,     // 只取顶层评论
        ...(chunkId ? { chunkId } : { chunkId: null }), // 文档级 or 段落级
      },
      include: {
        user: { select: { id: true, nickname: true, avatar: true } },
        replies: {
          include: {
            user: { select: { id: true, nickname: true, avatar: true } },
          },
          orderBy: { createdAt: 'asc' },
        },
      },
      orderBy: { createdAt: 'desc' },
    });
    return comments;
  }

  /** 解决批注 */
  async resolve(commentId: string, userId: string) {
    return this.prisma.comment.update({
      where: { id: commentId },
      data: { isResolved: true },
    });
  }

  /** 删除评论 */
  async remove(commentId: string, userId: string) {
    const comment = await this.prisma.comment.findUnique({ where: { id: commentId } });
    if (!comment) throw new BadRequestException('评论不存在');
    // 只有作者或管理员可以删除
    if (comment.userId !== userId) {
      const isAdmin = await this.accessControl.isSystemAdmin(userId);
      if (!isAdmin) throw new ForbiddenException('无权限删除');
    }
    return this.prisma.comment.delete({ where: { id: commentId } });
  }
}
```
