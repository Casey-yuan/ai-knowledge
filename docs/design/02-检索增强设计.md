## 设计文档：检索增强

版本：v1.0 | 日期：2026-06-11

---

### 一、现状分析

当前 `RetrievalService`（retrieval.service.ts）实现了三种检索方式：

1. **向量检索** `hybridSearch`：pgvector 余弦相似度，实际上只做向量检索
2. **全文检索** `fullTextSearch`：PostgreSQL `to_tsvector('chinese')` 全文搜索
3. **多知识库向量检索** `hybridSearchMultiple`：支持 `ANY($2)` 多知识库过滤

存在的问题：

1. **Rerank 未实现**：`KnowledgeBase.rerankEnabled` 字段存在但没有任何代码读取它
2. **查询重写未接入**：`LlmService.rewriteQuery` 已实现但 `ChatService` 未调用
3. **无混合检索融合**：向量检索和全文检索是分开的，`hybridSearch` 命名误导——实际只做向量检索
4. **无检索缓存**：相同查询每次都重新调用 embedding API 和数据库查询
5. **无向量索引**：未创建 HNSW 或 IVFFlat 索引，数据量大时性能堪忧
6. **无元数据过滤**：不能按标签、时间、文档类型等条件筛选检索结果

---

### 二、目标

1. 实现真正的混合检索（向量 + 全文 + RRF 融合排序）
2. 接入 Rerank 二次排序
3. 接入查询重写（对话场景）
4. 支持元数据过滤条件
5. 实现检索结果缓存
6. 创建向量索引优化查询性能

### 四、模块设计

#### 4.1 检索流水线重构

整体检索流程从"单一检索"升级为"多阶段流水线"：

```
用户查询
  ↓
[Step 1] 查询预处理
  ├─ 查询重写（对话场景，基于历史上下文）
  └─ 查询缓存检查
  ↓
[Step 2] 多路召回
  ├─ 向量检索（pgvector 余弦相似度）
  ├─ 全文检索（PostgreSQL tsvector）
  └─ 元数据过滤（标签、时间、类型等条件应用）
  ↓
[Step 3] 结果融合（RRF - Reciprocal Rank Fusion）
  ↓
[Step 4] Rerank 重排序（可选，基于知识库配置）
  ↓
[Step 5] 结果裁剪（按 topK 和 threshold 过滤）
  ↓
[Step 6] 缓存写入 & 返回
```

#### 4.2 混合检索 + RRF 融合

```typescript
// retrieval.service.ts 重构后的核心方法

async hybridSearchFused(
  kbIds: string[],
  query: string,
  options: {
    topK?: number;
    threshold?: number;
    vectorWeight?: number;   // 向量权重，默认 0.7
    fulltextWeight?: number; // 全文权重，默认 0.3
    filters?: RetrievalFilters;
  }
): Promise<RetrievalResult[]> {

  const { topK = 5, threshold = 0.3, vectorWeight = 0.7, fulltextWeight = 0.3 } = options;

  // 并行执行两路检索
  const [vectorResults, fulltextResults] = await Promise.all([
    this.vectorSearch(kbIds, query, topK * 3, threshold), // 多召回，留给融合排序裁剪
    this.fullTextSearchMultiple(kbIds, query, topK * 3),
  ]);

  // RRF 融合
  const fused = this.reciprocalRankFusion(
    vectorResults, fulltextResults,
    vectorWeight, fulltextWeight
  );

  // 裁剪
  return fused
    .filter(r => r.score >= threshold)
    .slice(0, topK);
}

/**
 * Reciprocal Rank Fusion
 * score = Σ weight_i / (k + rank_i)
 * k 通常取 60
 */
private reciprocalRankRankFusion(
  vectorResults: RetrievalResult[],
  fulltextResults: RetrievalResult[],
  vectorWeight: number,
  fulltextWeight: number,
  k: number = 60,
): RetrievalResult[] {
  const scoreMap = new Map<string, { score: number; result: RetrievalResult }>();

  vectorResults.forEach((r, rank) => {
    const rrfScore = vectorWeight / (k + rank + 1);
    const existing = scoreMap.get(r.id);
    if (existing) {
      existing.score += rrfScore;
    } else {
      scoreMap.set(r.id, { score: rrfScore, result: r });
    }
  });

  fulltextResults.forEach((r, rank) => {
    const rrfScore = fulltextWeight / (k + rank + 1);
    const existing = scoreMap.get(r.id);
    if (existing) {
      existing.score += rrfScore;
    } else {
      scoreMap.set(r.id, { score: rrfScore, result: r });
    }
  });

  return Array.from(scoreMap.values())
    .sort((a, b) => b.score - a.score)
    .map(v => ({ ...v.result, score: v.score }));
}
```

```typescript
// rerank.service.ts
@Injectable()
export class RerankService {
  /**
   * 对检索结果进行二次排序
   * 使用 DashScope 的 rerank 模型或简单的交叉编码器
   */
  async rerank(
    query: string,
    results: RetrievalResult[],
    topK: number
  ): Promise<RetrievalResult[]> {

    // 方案 A：调用 DashScope rerank API
    // 模型：gte-rerank
    const apiKey = process.env.ALIYUN_DASHSCOPE_API_KEY;
    const res = await axios.post(
      'https://dashscope.aliyuncs.com/api/v1/services/rerank',
      {
        model: 'gte-rerank',
        input: {
          query,
          documents: results.map(r => r.content),
        },
        parameters: { top_n: topK },
      },
      { headers: { Authorization: `Bearer ${apiKey}` } }
    );

    // 按 rerank 返回的分数重排
    const reranked = res.data.output.results;
    return reranked.map(r => ({
      ...results[r.index],
      rerankScore: r.relevance_score,
    }));
  }
}
```

#### 4.5 元数据过滤

```typescript
// retrieval.service.ts 新增过滤条件构建

interface RetrievalFilters {
  tags?: string[];           // 标签 ID 列表
  documentTypes?: string[];  // 文档类型：MARKDOWN, FILE
  mimeTypes?: string[];      // 文件 MIME 类型
  dateFrom?: Date;           // 文档创建时间范围
  dateTo?: Date;
  createdBy?: string;        // 文档创建者
}

private buildFilterClause(filters?: RetrievalFilters): { sql: string; params: any[] } {
  if (!filters) return { sql: '', params: [] };

  const clauses: string[] = [];
  const params: any[] = [];
  let paramIndex = 5; // 从 $5 开始，避免与现有参数冲突

  if (filters.tags?.length) {
    clauses.push(`EXISTS (
      SELECT 1 FROM "DocTag" dt
      WHERE dt."documentId" = d.id
      AND dt."tagId" = ANY($${paramIndex++})
    )`);
    params.push(filters.tags);
  }

  if (filters.documentTypes?.length) {
    clauses.push(`d.type = ANY($${paramIndex++})`);
    params.push(filters.documentTypes);
  }

  if (filters.dateFrom) {
    clauses.push(`d."createdAt" >= $${paramIndex++}`);
    params.push(filters.dateFrom);
  }

  if (filters.dateTo) {
    clauses.push(`d."createdAt" <= $${paramIndex++}`);
    params.push(filters.dateTo);
  }

  const sql = clauses.length > 0
    ? 'AND ' + clauses.join(' AND ')
    : '';

  return { sql, params };
}
```

#### 4.6 检索缓存

```typescript
// retrieval.service.ts 新增缓存逻辑

private async getCachedResults(
  kbIds: string[], query: string, topK: number, threshold: number
): Promise<RetrievalResult[] | null> {
  const queryHash = this.buildQueryHash(kbIds, query, topK, threshold);

  const cached = await this.prisma.retrievalCache.findUnique({
    where: { queryHash },
  });

  if (cached && cached.expiresAt > new Date()) {
    // 命中，增加计数
    await this.prisma.retrievalCache.update({
      where: { queryHash },
      data: { hitCount: { increment: 1 } },
    });
    return cached.results as RetrievalResult[];
  }

  // 过期，清理
  if (cached) {
    await this.prisma.retrievalCache.delete({ where: { queryHash } });
  }

  return null;
}

private async setCachedResults(
  kbIds: string[], query: string, topK: number, threshold: number,
  results: RetrievalResult[], ttlSeconds: number = 300 // 默认 5 分钟
): Promise<void> {
  const queryHash = this.buildQueryHash(kbIds, query, topK, threshold);
  const expiresAt = new Date(Date.now() + ttlSeconds * 1000);

  await this.prisma.retrievalCache.upsert({
    where: { queryHash },
    create: {
      kbIds: kbIds.sort().join(','),
      query, queryHash,
      results, expiresAt,
    },
    update: {
      results, expiresAt,
      hitCount: { increment: 1 },
    },
  });
}

private buildQueryHash(kbIds: string[], query: string, topK: number, threshold: number): string {
  const key = `${kbIds.sort().join(',')}:${query}:${topK}:${threshold}`;
  // 使用 Node.js crypto 生成短 hash
  return require('crypto').createHash('md5').update(key).digest('hex');
}
```
